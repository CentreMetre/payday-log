Table heists_completed {
  id int pk [increment]
  xp_amount int [not null]
  accurate_xp_input boolean [not null] // Whether it was a guess or not if the number wasnt seen
  heist int [not null, ref: > heists.id] // The heist itself, e.g. NRFTW, Road Rage etc.
  completed_at datetime [not null]
  heist_success boolean [not null]
  heist_finish_state int [not null] // Loud, stealth, alarm but pre assault
  all_bags_secured boolean [not null]
  difficulty int [not null, ref: > difficulties.id]
  notes text
}

Table heists {
  id int pk
  name text
}

Table difficulties {
  id int pk
  difficulty text
}

Table challenges_completed {
  id int pk
  challenge int [not null, ref: > challenges.id] // Challenge id
  completed boolean [not null]
  completed_at datetime
  notes text
}

Table challenges {
  id int pk
  challange text [not null]
}
